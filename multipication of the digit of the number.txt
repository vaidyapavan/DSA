package Basic_DSA;
import java.util.*;
public class Multipication_of_the_digit_of_number {

	public static void main(String[] args) {
		Scanner s= new Scanner(System.in);
		int no = s.nextInt();
		
		int mult=1;
		while(no>0)
		{
			int last_digit = no%10;
			mult= mult*last_digit;
			no = no/10;
			
		}
		
		System.out.println(mult);

	}

}

package Basic_DSA;
import java.util.*;
import java.util.Scanner;

public class Decimal_to_binary_conversion {
	
	 static void Decimal_To_binary(int no)
	{
		 
		 StringBuilder str = new StringBuilder();
		 if(no==0)
		 {
			 System.out.println(0);
			
		 }
		 
		 while(no>0)
		 {
			 int last_digit = no%2;
			 str.append( last_digit);
			 no = no/2;
		 }
		 
		 
	
		 
		 System.out.println(str);
		 
		
	}
	
	public static void main(String args[])
	{
		Scanner s = new Scanner(System.in);
		System.out.println("enter the bmber");
		int no = s.nextInt();
		Decimal_to_binary_conversion.Decimal_To_binary(no);
		
	}

}

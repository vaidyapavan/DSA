package Basic_DSA;
import java.util.*;

public class Removing_duplicate_from_Array {
	
	
	 static void removeduplicate()
	 {
		 int arr[]= {1,2,5,3,2,3,4,5};
		 ArrayList<Integer> myarr = new ArrayList<>();
		 int n= arr.length;
		 
		 for(int i=0;i<n;i++)
		 {
			 int element =arr[i];
			 if(!myarr.contains(element))
			 {
				 myarr.add(element);
				 
			 }
		 }
		 
		 System.out.println(myarr);
		 
		 
	 }

	public static void main(String[] args) {
		Removing_duplicate_from_Array.removeduplicate();
		

	}

}
